# Pruebas básicas de comandos
ls
echo "Hola mundo"
echo 'Comillas simples'
echo sin comillas
/bin/ls
/usr/bin/env

# Pruebas con pipes simples
ls | cat
echo "test" | cat
ls -l | grep ".c"

# Pipes múltiples
ls | cat | cat
echo "uno dos tres" | cat | cat | cat
ls -la | head | tail

# Redirections de input
cat < /etc/passwd
echo "test" > output.txt
cat < output.txt

# Redirections de output
echo "primera linea" > test.txt
echo "segunda linea" >> test.txt
cat test.txt

# Heredoc básico
cat << EOF
Esta es una prueba
de heredoc
EOF

# Heredoc con variables
cat << DELIMITER
El PATH es: $PATH
El status es: $?
DELIMITER

# Variables de entorno
echo $PATH
echo $HOME
echo $USER
echo "Mi PATH: $PATH"

# Variable $? (exit status)
true
echo $?
false
echo $?
ls /directorio/inexistente
echo $?

# Combinaciones complejas
echo "test" | cat > output.txt && cat output.txt
ls | grep ".c" | head -3
echo $PATH | cat | cat

# Comillas y expansión
echo "$PATH"
echo '$PATH'
echo "El usuario es: $USER"
echo 'El usuario es: $USER'

# Pipes con redirections
ls | cat > lista.txt
cat lista.txt
echo "nuevo contenido" >> lista.txt | cat

# Heredoc con pipes
cat << END | cat
Esto va por pipe
despues del heredoc
END

# Comandos con argumentos múltiples
echo uno dos tres cuatro cinco
ls -la /tmp
cat /etc/passwd | head -5 | tail -2

# Limpieza
rm -f output.txt test.txt lista.txt
exit
